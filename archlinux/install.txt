# パッケージのアップデート＆アップグレード
sudo pacman -Syyu

===== 環境のセットアップ ======
# fish shellの環境構築
sudo pacman -S fish

# Rust(Cargo)の環境構築
sudo pacman -S rustup # ~/.cargo/binをconfig.fishに

# Nodejs(nvm)の環境構築
sudo pacman -S nvm # nvmインストール
fisher install fabioantunes/fish-nvm # nvmのfish用環境を構築してくれるプラグインをインストール
fisher install edc/bass

# Nim(choosenim)の環境構築
curl https://nim-lang.org/choosenim/init.sh -sSf | sh # nim版pyenv的なchoosenim
