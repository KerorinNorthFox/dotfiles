# パッケージのアップデート＆アップグレード
sudo apt update && sudo apt upgrade -y

===== 環境のセットアップ ======
# fish shellの環境構築
sudo apt install fish 
sudo chsh $USER -s /usr/bin/fish # シェルの変更 // Authentication failureを回避するために, sudoで実行とユーザー指定
curl -sL https://raw.githubusercontent.com/jorgebucaran/fisher/main/functions/fisher.fish | source && fisher install jorgebucaran/fisher 

# Rust(Cargo)の環境構築
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh # ~/.cargo/binをconfig.fishに

# Nodejs(nvm)の環境構築
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/master/install.sh | bash # nvmインストール
fisher install jorgebucaran/nvm.fish

# Nim(choosenim)の環境構築
